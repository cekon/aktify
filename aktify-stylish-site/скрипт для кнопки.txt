async function payOrder() {
  // Данные заказа (формируются динамически на сайте)
  const orderData = {
    orderId: "ORDER-" + Math.floor(Math.random() * 100000), // Уникальный номер заказа
    sku: "1695005157",                                     // SKU товара
    quantity: 1                                             // Количество
  };

  try {
    // Вызываем ваш Cloudflare Worker
    const response = await fetch('https://workers.dev', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(orderData)
    });

    const data = await response.json();

    if (data.payment_link) {
      // Перенаправляем пользователя на платежную форму Ozon Pay
      window.location.href = data.payment_link;
    } else {
      alert('Ошибка генерации ссылки: ' + JSON.stringify(data));
    }
  } catch (error) {
    console.error('Ошибка сети:', error);
  }
}
