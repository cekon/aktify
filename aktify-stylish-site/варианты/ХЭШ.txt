// Функция генерации HMAC-SHA256 подписи в формате Hex
async function generateHmacSha256(message, secretKey) {
  const encoder = new TextEncoder();
  
  // 1. Превращаем секретный ключ и сообщение в байтовые массивы (Uint8Array)
  const keyData = encoder.encode(secretKey);
  const messageData = encoder.encode(message);

  // 2. Импортируем сырой ключ в криптографический объект Cloudflare
  const cryptoKey = await crypto.subtle.importKey(
    "raw",
    keyData,
    { name: "HMAC", hash: "SHA-256" },
    false,
    ["sign"]
  );

  // 3. Подписываем сообщение алгоритмом HMAC
  const signatureBuffer = await crypto.subtle.sign("HMAC", cryptoKey, messageData);

  // 4. Переводим полученный Buffer в строку Hex (шестнадцатеричный вид)
  const signatureArray = Array.from(new Uint8Array(signatureBuffer));
  const signatureHex = signatureArray
    .map(b => b.toString(16).padStart(2, "0"))
    .join("");

  return signatureHex;
}
