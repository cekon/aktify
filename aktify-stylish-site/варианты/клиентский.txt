<!-- index.html -->
<button id="payButton">Купить за 1500 ₽</button>

<script>
document.getElementById('payButton').addEventListener('click', async () => {
    // Блокируем кнопку от повторных кликов
    const btn = document.getElementById('payButton');
    btn.disabled = true;
    btn.innerText = 'Оформление...';

    // Тестовые данные текущего заказа
    const orderData = {
        orderNumber: 'INV-' + Date.now(), // Генерируем уникальный номер заказа
        amount: 1500.00,
        userEmail: 'customer@example.com'
    };

    try {
        // Отправляем запрос на НАШ сервер (из Шага 1)
        const response = await fetch('http://localhost:3000/api/create-payment', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(orderData)
        });

        const result = await response.json();

        if (result.success && result.paymentUrl) {
            // Главный момент: Перенаправляем пользователя на платежную страницу Ozon Pay
            window.location.href = result.paymentUrl;
        } else {
            alert('Ошибка при создании платежа: ' + result.error);
            btn.disabled = false;
            btn.innerText = 'Купить';
        }
    } catch (error) {
        alert('Не удалось связаться с сервером.');
        btn.disabled = false;
        btn.innerText = 'Купить';
    }
});
</script>
